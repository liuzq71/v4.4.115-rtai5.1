76 // proc print macros - Contributed by: Erwin Rol (erwin@muffin.org)

78 // macro that holds the local variables that
79 // we use in the PROC_PRINT_* macros. We have
80 // this macro so we can add variables with out
81 // changing the users of this macro, of course
82 // only when the names don't colide!
83 #define PROC_PRINT_VARS \
84  off_t pos = 0; \
85  off_t begin = 0; \
86  int len = 0 /* no ";" */
87 
88 // macro that prints in the procfs read buffer.
89 // this macro expects the function arguments to be
90 // named as follows.
91 // static int FOO(char *page, char **start,
92 // off_t off, int count, int *eof, void *data)
93 
94 #define PROC_PRINT(fmt,args...) \
95 do { \
96  len += sprintf(page + len , fmt, ##args); \
97  pos += len; \
98  if(pos < off) { \
99  len = 0; \
100  begin = pos; \
101  } \
102  if(pos > off + count) \
103  goto done; \
104 } while(0)
105 
106 // macro to leave the read function for a other
107 // place than at the end.
108 #define PROC_PRINT_RETURN \
109 do { \
110  *eof = 1; \
111  goto done; \
112 } while(0)
113 
114 // macro that should only used ones at the end of the
115 // read function, to return from a other place in the
116 // read function use the PROC_PRINT_RETURN macro.
117 #define PROC_PRINT_DONE \
118 do { \
119  *eof = 1; \
120  done: \
121  *start = page + (off - begin); \
122  len -= (off - begin); \
123  if(len > count) \
124  len = count; \
125  if(len < 0) \
126  len = 0; \
127  return len; \
128 } while(0)
129
